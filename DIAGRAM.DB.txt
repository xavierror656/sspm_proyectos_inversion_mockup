// DBML compatible con https://dbdiagram.io
// Submódulo: Cartera, Seguimiento y Análisis de Proyectos de Inversión SSPM
// Actualizado al flujo validado:
// - En Cartera no se suben documentos.
// - La carga documental vive en Proyecto Adquisitivo / Compras.
// - Al capturar el ID del proyecto de inversión en Proyecto Adquisitivo, cartera queda committed.
// - Al formalizar la referencia adquisitiva, cartera queda formalized y se actualiza committed_amount.
// Convención:
// - Tables, fields, enums and relationships in English.
// - Notes in Spanish.
// - References prefixed with external_ represent existing platform modules.
// - Own module tables do not use external_ prefix.
// - Objects prefixed with vw_ represent SQL views or materialized views for analysis.

Project SSPM_Investment_Portfolio_With_Analysis {
  database_type: "PostgreSQL"
  Note: "Modelo conceptual actualizado para Cartera, Seguimiento y Análisis de Proyectos de Inversión SSPM. Cartera no almacena carga documental directa; los documentos se cargan en Proyecto Adquisitivo y se consultan desde cartera por vínculo al ID del proyecto de inversión."
}

Table investment_projects {
  id bigserial [pk]
  internal_folio varchar(30) [not null, unique, note: "Formato CART-0001-2026. Automático e inmutable."]
  municipal_folio varchar(80) [note: "Opcional. Captura manual si Municipio asigna folio."]
  name varchar(250) [not null]
  description text
  objective text
  requesting_area varchar(180)
  fiscal_year int [not null]
  status investment_project_status [not null, default: "initial_registration"]

  initial_authorized_amount decimal(14,2) [not null, default: 0]
  current_amount decimal(14,2) [not null, default: 0, note: "No se edita manualmente. Se recalcula por movimientos."]
  committed_amount decimal(14,2) [not null, default: 0, note: "Suma de financial_commitments formalizados desde Proyecto Adquisitivo."]
  remaining_balance decimal(14,2) [not null, default: 0, note: "current_amount - committed_amount."]

  is_multi_year boolean [not null, default: false]
  start_date date
  end_date date
  budget_line varchar(80)
  sub_account varchar(80)
  notes text
  closed_at timestamp

  created_by_user_id bigint [not null]
  updated_by_user_id bigint
  created_by_person_id bigint [note: "Duplicado controlado para trazabilidad con RH si la arquitectura lo requiere."]
  updated_by_person_id bigint
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  indexes {
    internal_folio [unique]
    fiscal_year
    status
    is_multi_year
    created_by_user_id
    created_by_person_id
  }

  Note: "Entidad raíz del submódulo. Controla autorización, monto vigente, compromiso, remanente, estado, plurianualidad y cierre formal. No debe recibir carga documental directa desde Cartera."
}

Table project_movements {
  id bigserial [pk]
  investment_project_id bigint [not null]
  movement_type project_movement_type [not null]
  movement_at timestamp [not null, default: `now()`]

  amount decimal(14,2) [not null, default: 0]
  sign smallint [not null, default: 0, note: "1 suma, -1 resta, 0 sin afectación financiera."]
  previous_current_amount decimal(14,2)
  new_current_amount decimal(14,2)

  responsible_user_id bigint [not null]
  responsible_person_id bigint
  approved_by_user_id bigint
  approved_by_person_id bigint

  redistribution_id bigint
  notes text
  created_at timestamp [not null, default: `now()`]

  indexes {
    investment_project_id
    movement_type
    movement_at
    responsible_user_id
    responsible_person_id
    redistribution_id
  }

  Note: "Todo cambio financiero o administrativo del proyecto debe registrarse aquí. El monto vigente se actualiza mediante estos movimientos, nunca por edición directa."
}

Table project_redistributions {

    redistribution_folio [unique]

    target_project_id

  Note: "Genera doble asiento: salida en proyecto origen y entrada en proyecto destino. Debe validar que el monto no exceda el remanente del origen."

Table financial_commitments {
  id bigserial [pk]
  investment_project_id bigint [not null]

  reference_type procurement_reference_type [not null, note: "requisition, contract o purchase_order."]
  reference_id bigint [not null, note: "ID de la tabla real en Compras. Puede apuntar a requisiciones, contratos o pedidos según reference_type."]
  reference_folio varchar(100)

  linked_at timestamp [not null, default: `now()`]
  formalized_at timestamp
  committed_amount decimal(14,2) [not null, default: 0]
  status financial_commitment_status [not null, default: "linked"]

  linked_by_user_id bigint [not null]
  linked_by_person_id bigint
  notes text
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  indexes {
    investment_project_id
    (reference_type, reference_id) [unique]
    status
    formalized_at
    linked_by_user_id
  }

  Note: "Tabla puente flexible para no acoplar el módulo a una única tabla de Compras. Al insertar vínculo por ID, investment_projects.status pasa a committed. Cuando la referencia se formaliza, pasa a formalized y actualiza committed_amount."
}
